# Code Map: tcart.js

## Module Overview

Файл tcart.js (9104 строки) — скомпилированный Babel-бандл модуля корзины покупок платформы **Tilda** (блок ST100 / t706). Содержит полный цикл работы корзины: инициализацию, управление товарами, скидки, промокоды, авторизацию через Members, восстановление потерянных корзин, отрисовку UI (иконка, sidebar, fullscreen), работу с формой заказа, доставку и сетевое взаимодействие с Tilda Store API. ~42% файла занимают Babel runtime helpers.

## ⚠️ Unclear / Needs Review

1. **`tcart__addEvent__links` (~400 строк)** — слишком длинная функция, обрабатывающая клики по `#order` ссылкам. Внутри может быть сложная логика маршрутизации, которая требует детальной трассировки.
2. **`tcart__init`** — главная точка входа, но её точная последовательность вызовов не полностью трассирована (зависит от множества условий: авторизация, доставка, скидки, lost cart).
3. **`tcart__updateTotalProductsinCartObj`** — пересчёт сумм корзины, может содержать нетривиальную логику промокодов/скидок.
4. **Опечатка**: `t_cart__loadСountryСodes` — кириллические «С» вместо латинских «C» в имени функции.
5. **Глобальное состояние**: `window.tcart` используется повсеместно как глобальный стейт — нет модульной инкапсуляции.
6. **Взаимодействие с внешними модулями**: ожидание загрузки `tcart__calcAmountWithDiscounts`, `tcart_newDelivery`, `tcart__restoreFields` через `t_store__useExternalFunction` — фактическое поведение зависит от внешних скриптов.

import type { PickupPointInfoItem, TildaPickupPoint } from './types';

export function mapToTildaPickupPoint(
  point: PickupPointInfoItem,
): TildaPickupPoint | null {
  if (!point.enabled) return null;

  const dm = point.delivery_method;
  const { lat, long } = dm.coordinates;

  // Формируем строку режима работы из массива working_hours
  const workTime = dm.working_hours
    .slice(0, 1) // берём первый день как образец
    .flatMap(wh =>
      wh.periods.map(
        p =>
          `${String(p.min.hours).padStart(2, '0')}:${String(p.min.minutes).padStart(2, '0')}` +
          `–` +
          `${String(p.max.hours).padStart(2, '0')}:${String(p.max.minutes).padStart(2, '0')}`,
      ),
    )
    .join(', ') || '';

  return {
    id: String(dm.map_point_id),
    name: dm.name,
    address: dm.address,
    coordinates: [lat, long],
    workTime,
    phones: [],        // Ozon не отдаёт телефоны в этом методе
    addressComment: dm.description ?? '',
    cash: 'n',         // Ozon не принимает наличные
    postalCode: '',    // нет в ответе
  };
}

import { getAccessToken } from '../../utils/getAccessToken';
import { ozonConfig } from '../../config/env';
import { OzonApiClient } from '../../utils/typedApiClient';

export interface DeliveryPriceRequest {
  mapPointId: number;
  items: { sku: number; quantity: number; offer_id: string }[];
  buyerPhone?: string;
}

export interface DeliveryPriceResult {
  price: number;
  currency: string;
  daysMin: number;
  daysMax: number;
}

export async function getDeliveryPrice(
  req: DeliveryPriceRequest,
): Promise<DeliveryPriceResult> {
  const token = getAccessToken();
  if (!token) throw new Error('Требуется авторизация');

  const client = new OzonApiClient(ozonConfig.apiUrl, token, ozonConfig.timeoutMs);

  // /v2/delivery/checkout — проверяем доступность и получаем цену
  const response = await fetch(`${ozonConfig.apiUrl}/v2/delivery/checkout`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${token}`,
    },
    body: JSON.stringify({
      delivery_type: {
        pick_up: { map_point_id: req.mapPointId },
      },
      items: req.items,
    }),
  });

  if (!response.ok) {
    const err = await response.json().catch(() => ({}));
    throw new Error(`Ozon checkout error: ${JSON.stringify(err)}`);
  }

  const data = await response.json();

  // Берём первый доступный сплит
  const split = data.splits?.find(
    (s: any) => s.unavailable_reason === 'UNSPECIFIED',
  );

  if (!split) throw new Error('Доставка недоступна для этой точки');

  const timeslot = split.delivery_method?.timeslots?.[0];

  return {
    price: 0,      // /v2/checkout не возвращает цену — нужен /v2/order/create
    currency: 'RUB',
    daysMin: timeslot
      ? daysBetween(new Date(), new Date(timeslot.client_date_range.from))
      : 0,
    daysMax: timeslot
      ? daysBetween(new Date(), new Date(timeslot.client_date_range.to))
      : 0,
  };
}

function daysBetween(a: Date, b: Date): number {
  return Math.ceil((b.getTime() - a.getTime()) / (1000 * 60 * 60 * 24));
}

(function () {
  const BACKEND = 'https://your-backend.com';

  function waitFor(getter, cb, ms = 200, max = 75) {
    let n = 0;
    const t = setInterval(() => {
      const val = getter();
      if (val) { clearInterval(t); cb(val); }
      if (++n >= max) clearInterval(t);
    }, ms);
  }

  // Текущий запрос — отменяем предыдущий если карта двигается быстро
  let currentRequest = null;

  // Загрузка точек по viewport карты
  async function loadPointsForViewport(map, delivery) {
    const bounds = map.getBounds(); // [[lat1,lng1],[lat2,lng2]]

    // Отменяем предыдущий запрос
    if (currentRequest) {
      currentRequest.cancelled = true;
    }
    const req = { cancelled: false };
    currentRequest = req;

    try {
      const res = await fetch(`${BACKEND}/delivery/map`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          viewport: {
            left_bottom:  { lat: bounds[0][0], long: bounds[0][1] },
            right_top:    { lat: bounds[1][0], long: bounds[1][1] },
          },
          zoom: map.getZoom(),
        }),
      });

      if (req.cancelled) return; // карта уже уехала дальше

      const data = await res.json(); // массив TildaPickupPoint[]
      delivery.mapAddPoints(data);
    } catch (e) {
      console.warn('[OzonPatch] ошибка загрузки точек:', e);
    }
  }

  waitFor(() => window.tcart_newDelivery, (delivery) => {

    // 1. Перехватываем mapInit — вешаем boundschange после того как карта создана
    const origMapInit = delivery.mapInit?.bind(delivery);
    delivery.mapInit = function (...args) {
      const result = origMapInit?.(...args);

      // mapInit создаёт карту и пишет её в window.t_delivery__map
      waitFor(() => window.t_delivery__map, (map) => {
        console.log('[OzonPatch] карта найдена, вешаем boundschange');

        // debounce — не долбим бэк при каждом пикселе
        let debounceTimer = null;
        map.events.add('boundschange', () => {
          clearTimeout(debounceTimer);
          debounceTimer = setTimeout(() => {
            loadPointsForViewport(map, delivery);
          }, 400);
        });

        // Загружаем точки для начального viewport сразу
        loadPointsForViewport(map, delivery);
      });

      return result;
    };

    // 2. getPickupList — отдаём ПУСТОЙ массив
    // Точки на карте грузим сами через boundschange
    // Searchbox (поиск текстом) — отдаём из кэша по pattern
    delivery.getPickupList = function (params) {
      if (params.pattern) {
        // Текстовый поиск — идём на бэк
        fetch(`${BACKEND}/delivery/pvz?pattern=${encodeURIComponent(params.pattern)}&postalCode=${params.postalCode ?? ''}`)
          .then(r => r.json())
          .then(data => params.onDone(data))
          .catch(() => params.onFail?.());
      } else {
        // Начальная загрузка без поиска — карта сама загрузит через boundschange
        params.onDone([]);
      }
    };

    console.log('[OzonPatch] Патч применён ✓');
  });

          .then(data => params.onDone(data.pvz ?? []))
          .catch(() => params.onFail?.());
      } else {
        // Инициализация карты → пустой массив, точки грузим сами
        params.onDone([]);
      }
    };

    console.log('[OzonPatch] Патч применён ✓');
  });

})();
Два момента которые уточнились из логов:

'pattern' in params вместо params.pattern !== undefined — надёжнее, потому что ключ физически отсутствует в объекте при вызове карты, а не просто undefined.

mapAddPoints(null, points, null) — теперь знаем что второй аргумент это массив точек. null для первого и третьего — безопаснее чем undefined, но нужно проверить живым запуском. Если упадёт — покажи ошибку и разберём что ожидается в t и n.

бля, почему запросы по поиску точек в бд хуярят до 100% cpu

SQLite + LIKE с % в начале строки — это full table scan, индексы не используются вообще.

Решение — FTS5 (Full Text Search), встроен в SQLite, Bun поддерживает.

Шаг 3 — заменить запрос в getTildaPickupPoints
typescript
export function getTildaPickupPoints(options?: {
  q?: string;
  limit?: number;
  offset?: number;
}): {
  pvz: TildaPickupPoint[];
  total: number;
  limit: number;
  offset: number;
} {
  const { q, limit = 500, offset = 0 } = options || {};

  let rows: PickupPointRow[];
  let total: number;

  if (q) {
    // FTS поиск — быстро, индекс
    // * в конце = prefix search: "авто" найдёт "автово"
    const ftsQuery = q.trim().split(/\s+/).map(w => `${w}*`).join(' ');

    const ftsRows = db.prepare(`
      SELECT p.*
      FROM pickup_points p
      JOIN pickup_points_fts f ON f.map_point_id = p.map_point_id
      WHERE pickup_points_fts MATCH ?
        AND p.enabled = 1
      ORDER BY rank
      LIMIT ? OFFSET ?
    `).all(ftsQuery, limit, offset) as PickupPointRow[];

    const ftsTotal = db.prepare(`
      SELECT COUNT(*) as total
      FROM pickup_points p
      JOIN pickup_points_fts f ON f.map_point_id = p.map_point_id
      WHERE pickup_points_fts MATCH ?
        AND p.enabled = 1
    `).get(ftsQuery) as { total: number };

    rows = ftsRows;
    total = ftsTotal.total;
  } else {
    // Без поиска — обычная пагинация
    rows = db.prepare(`
      SELECT * FROM pickup_points
      WHERE enabled = 1
      ORDER BY city, name
      LIMIT ? OFFSET ?
    `).all(limit, offset) as PickupPointRow[];

    total = (db.prepare(
      'SELECT COUNT(*) as total FROM pickup_points WHERE enabled = 1'
    ).get() as { total: number }).total;
  }

  return { pvz: rows.map(mapRowToTilda), total, limit, offset };
}
Почему это быстро
LIKE с % в начале — O(n), читает каждую строку. FTS5 строит инвертированный индекс (слово → список id) — поиск O(log n), практически мгновенный даже на 85к строк. rank — встроенная релевантность FTS5, самые точные совпадения идут первыми.

После добавления FTS нужно удалить старую БД и дать ей пересоздаться, либо вызвать rebuildFts() вручную один раз — триггеры начнут работать только для новых записей.
